def add_comment
              @comment = Comment.new(params[:comment])
              @post = Post.find(params[:id])
              @comment.post = @post
              @comment.status = "new"
              if @comment.save
                      flash[:notice] = 'Your comment was submitted successfully.'
                      redirect_to(:action => 'view_post', :id => @post.id)
              else
    render('view_post')
              end
end
